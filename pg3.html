Simple Interface for shapes
import "math"

type Shape interface {
Area() float64
}

type Square struct { // Note: no "implements" declaration, needs to be type float64 because of the math import math.Pow function
side float64
}

func (sq Square) Area() float64 { return sq.side * sq.side }

type Circle struct { // No "implements" declaration here either, needs to be type float64 because of the math import math.Pow function
radius float64
}

func (c Circle) Area() float64 { return math.Pi * math.Pow(c.radius, 2) }
